/**
 * Represents the type of data encoded by a barcode -- from plain text, to a

 *
 * @author Sean Owen
 */

  ADDRESSBOOK,
  EMAIL_ADDRESS,
  PRODUCT,
  URI,
  TEXT,
  GEO,
  TEL,
  SMS,
  CALENDAR,
  WIFI,
  ISBN,
  VIN,

}
